// Function to draw a box
void drawBox(int width, int height)
 {

    for
	 (int i = 0; i < width; i++) 
        cout << "*";

    cout << endl;


    for 
	(int i = 0; i < height - 2; i++) {

        cout << "*";


        for
		 (int j = 0; j < width - 2; j++) 
            cout << " ";

        cout << "*\n";

    }

    for 
	(int i = 0; i < width; i++) 
        cout << "*";

    cout << endl;

}

// Function to calculate BMI
float calculateBMI(float weight, float height) {

    return weight / (height * height);

}
